<!-- operators in find() --->

db.collection.find( {marks : {$gt: 75}} )  // $gt is the greater-than 75  

db.collection.find( {city: { $in: ["hyd" , "knl"]}} )   // $in find to search any one hyd or knl or both

db.collection.find( {$or: [ {marks: {$gt: 75}, {city: "HYD"}} ]} )  // $or  is and OR operator
